# 模块化开发

> 本章节主要讲解前端模块化开发的核心理念与实践，包括模块化规范、代码组织、依赖管理等内容。

## 2.1 模块化规范

### 2.1.1 模块化的发展历程

模块化是前端工程化的基础，它经历了从无到有、从简单到复杂的发展过程。

**早期的全局变量模式**
在早期，JavaScript没有模块化机制，开发者通常使用全局变量来共享代码：

```javascript
// module1.js
var name = 'Module1';

function sayHello() {
  console.log('Hello from ' + name);
}

// 使用全局变量
window.Module1 = {
  name: name,
  sayHello: sayHello
};
```

```javascript
// main.js
Module1.sayHello(); // "Hello from Module1"
```

这种方式的缺点：
- 命名冲突：不同模块可能使用相同的变量名
- 依赖混乱：无法明确模块之间的依赖关系
- 难以维护：模块之间的关系不清晰

**IIFE（立即调用函数表达式）**
IIFE是一种早期的模块化解决方案，通过创建函数作用域来避免全局污染：

```javascript
// module.js
var Module = (function() {
  var _private = 'I am private';

  function _privateMethod() {
    console.log(_private);
  }

  return {
    publicMethod: function() {
      _privateMethod();
    }
  };
})();

Module.publicMethod(); // "I am private"
console.log(Module._private); // undefined
```

```javascript
// 模块依赖
var ModuleA = (function() {
  return {
    name: 'Module A'
  };
})();

var ModuleB = (function(ModuleA) {
  var name = 'Module B';

  function getFullName() {
    return name + ' depends on ' + ModuleA.name;
  }

  return {
    getFullName: getFullName
  };
})(ModuleA);

console.log(ModuleB.getFullName()); // "Module B depends on Module A"
```

### 2.1.2 CommonJS规范

CommonJS是Node.js使用的模块化规范，它使用`require`来引入模块，`module.exports`来导出模块。

**基本用法**
```javascript
// math.js - 导出模块
function add(a, b) {
  return a + b;
}

function subtract(a, b) {
  return a - b;
}

function multiply(a, b) {
  return a * b;
}

function divide(a, b) {
  if (b === 0) {
    throw new Error('Cannot divide by zero');
  }
  return a / b;
}

// 导出方式1：导出多个
module.exports = {
  add,
  subtract,
  multiply,
  divide
};
```

```javascript
// index.js - 引入模块
const math = require('./math');

console.log(math.add(1, 2)); // 3
console.log(math.subtract(5, 3)); // 2
console.log(math.multiply(3, 4)); // 12
console.log(math.divide(10, 2)); // 5
```

**模块导出函数或类**
```javascript
// logger.js
class Logger {
  constructor(prefix) {
    this.prefix = prefix;
  }

  log(message) {
    console.log(`[${this.prefix}] ${message}`);
  }

  error(message) {
    console.error(`[${this.prefix}] ERROR: ${message}`);
  }

  warn(message) {
    console.warn(`[${this.prefix}] WARN: ${message}`);
  }
}

module.exports = Logger;
```

**模块导出单例**
```javascript
// store.js
let count = 0;

function increment() {
  count++;
  return count;
}

function decrement() {
  count--;
  return count;
}

function getCount() {
  return count;
}

// 导出函数而非对象，每次require返回的都是同一个模块实例
module.exports = {
  increment,
  decrement,
  getCount
};
```

### 2.1.3 AMD规范

AMD（Asynchronous Module Definition）是异步模块定义规范，主要用于浏览器环境，解决了CommonJS在浏览器中同步加载的问题。

**基本用法**
```javascript
// 定义模块
define('myModule', ['dependency1', 'dependency2'], function(dep1, dep2) {
  // 模块逻辑
  function doSomething() {
    return dep1.getValue() + dep2.getValue();
  }

  // 导出
  return {
    doSomething: doSomething
  };
});

// 引入模块
require(['myModule', 'simpleModule'], function(myModule, simpleModule) {
  console.log(myModule.doSomething());
  console.log(simpleModule.name);
});
```

### 2.1.4 CMD规范

CMD（Common Module Definition）是SeaJS推广的模块定义规范，与AMD类似但有所不同。CMD推崇依赖就近，延迟执行。

**基本用法**
```javascript
// SeaJS示例
// math.js
define(function(require, exports, module) {
  var add = function(a, b) {
    return a + b;
  };

  var subtract = function(a, b) {
    return a - b;
  };

  exports.add = add;
  exports.subtract = subtract;
});
```

**AMD与CMD的区别**
```javascript
// AMD - 依赖前置
define(['dep1', 'dep2', 'dep3'], function(dep1, dep2, dep3) {
  // 所有依赖在定义时就已经加载
  dep1.doSomething();
  dep2.doSomething();
});

// CMD - 依赖就近
define(function(require, exports, module) {
  var dep1 = require('dep1'); // 使用时再加载
  dep1.doSomething();

  if (condition) {
    var dep2 = require('dep2'); // 条件加载
    dep2.doSomething();
  }
});
```

### 2.1.5 ES Module规范

ES Module（ESM）是ES6引入的官方模块化规范，使用`import`和`export`关键字。

**基本导出方式**

**命名导出**
```javascript
// named-export.js
// 方式1：声明时导出
export const name = 'Module';
export const version = '1.0.0';

export function add(a, b) {
  return a + b;
}

export class Calculator {
  constructor() {
    this.result = 0;
  }

  add(n) {
    this.result += n;
    return this;
  }

  subtract(n) {
    this.result -= n;
    return this;
  }

  getResult() {
    return this.result;
  }
}
```

**默认导出**
```javascript
// default-export.js
export default function() {
  return 'Default export';
}
```

**基本导入方式**

**命名导入**
```javascript
// import-named.js
import { name, version, add, Calculator } from './named-export.js';

console.log(name); // 'Module'
console.log(version); // '1.0.0'
console.log(add(1, 2)); // 3
```

**默认导入**
```javascript
// import-default.js
import defaultExport from './default-export.js';

console.log(defaultExport()); // 'Default export'
```

**命名空间导入**
```javascript
// import-namespace.js
import * as module from './named-export.js';

console.log(module.name); // 'Module'
console.log(module.add(1, 2)); // 3
```

**动态导入**
```javascript
// dynamic-import.js
// 返回Promise
import('./module.js')
  .then(module => {
    console.log(module.default);
    console.log(module.namedExport);
  });

// 使用async/await
async function loadModule() {
  const module = await import('./module.js');
  return module;
}
```

**ES Module的高级特性**

**静态导入与导出**
ES Module使用静态语法，导入导出必须在模块顶层，这使得：
- 静态分析成为可能
- Tree Shaking可以移除未使用的代码
- 打包时可以确定依赖关系

**ES Module在浏览器中的使用**
```html
<!-- index.html -->
<script type="module">
  import { createApp } from './app.js';

  createApp();
</script>
```

### 2.1.6 UMD规范

UMD（Universal Module Definition）是一种兼容多种模块化规范的模式，可以在CommonJS、AMD和全局变量环境中使用。

**UMD模式**
```javascript
// umd-module.js
(function(root, factory) {
  // 检测环境
  if (typeof module === 'object' && module.exports) {
    // CommonJS
    module.exports = factory();
  } else if (typeof define === 'function' && define.amd) {
    // AMD
    define([], factory);
  } else {
    // 全局变量
    root.MyModule = factory();
  }
}(typeof self !== 'undefined' ? self : this, function() {
  // 模块实现
  return {
    name: 'My Module',
    version: '1.0.0',
    init: function() {
      console.log('Module initialized');
    }
  };
}));
```

### 2.2.2 模块与组件的设计原则

**SOLID原则在前端中的应用**

**单一职责原则（SRP）**
每个模块、组件只负责一项功能。

```javascript
// 正确：分离职责
// validators/user.js
export function validateUser(user) {
  if (!user.name || !user.email) {
    throw new Error('Invalid user data');
  }
  return true;
}

// formatters/user.js
export function formatUser(user) {
  return {
    name: user.name.trim(),
    email: user.email.toLowerCase()
  };
}

// services/user.js
export async function saveUser(user) {
  const response = await fetch('/api/users', {
    method: 'POST',
    body: JSON.stringify(user)
  });
  return response.json();
}

// 使用
import { validateUser } from './validators/user';
import { formatUser } from './formatters/user';
import { saveUser } from './services/user';

async function createUser(userData) {
  validateUser(userData);
  const formattedUser = formatUser(userData);
  return saveUser(formattedUser);
}
```

**开闭原则（OCP）**
对扩展开放，对修改封闭。

```javascript
// 正确：使用策略模式
// 支付策略接口
class PaymentStrategy {
  pay(amount) {
    throw new Error('Method not implemented');
  }
}

// 具体支付策略
class AlipayStrategy extends PaymentStrategy {
  pay(amount) {
    console.log(`Alipay: ${amount}`);
  }
}

class WechatPayStrategy extends PaymentStrategy {
  pay(amount) {
    console.log(`Wechat Pay: ${amount}`);
  }
}

// 支付处理器
class PaymentProcessor {
  constructor(strategy) {
    this.strategy = strategy;
  }

  setStrategy(strategy) {
    this.strategy = strategy;
  }

  processPayment(amount) {
    return this.strategy.pay(amount);
  }
}
```

**依赖倒置原则（DIP）**
依赖抽象，不依赖具体实现。

```javascript
// 正确：依赖抽象
class UserRepository {
  save(user) {
    throw new Error('Method not implemented');
  }
}

class MySQLUserRepository extends UserRepository {
  save(user) {
    console.log('Saving to MySQL');
  }
}

class UserService {
  constructor(repository) {
    this.repository = repository;
  }

  createUser(user) {
    this.repository.save(user);
  }
}
```

## 2.3 依赖管理

### 2.3.1 依赖管理基础

**package.json详解**
```json
{
  "name": "my-project",
  "version": "1.0.0",
  "description": "My awesome project",
  "type": "module",
  "main": "dist/index.js",
  "module": "dist/index.esm.js",
  "types": "dist/index.d.ts",
  "exports": {
    ".": {
      "import": "./dist/index.esm.js",
      "require": "./dist/index.cjs.js",
      "types": "./dist/index.d.ts"
    }
  },
  "scripts": {
    "dev": "vite",
    "build": "vite build",
    "preview": "vite preview",
    "lint": "eslint src --ext .vue,.js,.jsx,.ts,.tsx",
    "test": "jest"
  },
  "dependencies": {
    "vue": "^3.3.4",
    "vue-router": "^4.2.4",
    "pinia": "^2.1.6",
    "axios": "^1.5.0"
  },
  "devDependencies": {
    "@vitejs/plugin-vue": "^4.3.4",
    "vite": "^4.4.9",
    "eslint": "^8.49.0"
  }
}
```

**依赖类型详解**

dependencies（运行时依赖）：项目运行需要的依赖
devDependencies（开发时依赖）：只在开发时需要的依赖
peerDependencies（对等依赖）：期望使用者提供的依赖
optionalDependencies（可选依赖）：可选的依赖

### 2.3.2 版本管理

**语义化版本（SemVer）**

版本格式：`主版本.次版本.修订号`

```bash
# ^ 允许次版本和修订号升级
npm install lodash@^4.17.21
# 会安装 4.x.x 中的最新版本

# ~ 允许修订号升级
npm install lodash@~4.17.21
# 会安装 4.17.x 中的最新版本

# 精确版本
npm install lodash@4.17.21
```
